    <script>
        // 頁面背景用 bg.jpg；球場 Canvas 背景用 field.jpg
        const imgField = new Image(); imgField.src = 'field.jpg';
        const imgBall = new Image(); imgBall.src = 'ball.png';
        const imgShooter = new Image(); imgShooter.src = 'shooter.png';
        const imgGoalie = new Image(); imgGoalie.src = 'goalie.png';
        const imgBlock = new Image(); imgBlock.src = 'block.png'; 

        let empId = "";
        let today = new Date().toDateString(); 
        const MAX_PLAYS = 3; 
        
        function checkLogin() {
            empId = document.getElementById("emp-id").value.trim();
            if (empId === "") { alert("請輸入員工號碼！"); return; }
            
            let records = JSON.parse(localStorage.getItem("dailyRecords") || "{}");
            if (!records[today]) records[today] = {}; 
            
            let playCount = records[today][empId] || 0; 

            if (playCount >= MAX_PLAYS) {
                document.getElementById("login-error").style.display = "block";
            } else {
                records[today][empId] = playCount + 1;
                localStorage.setItem("dailyRecords", JSON.stringify(records));
                
                // 登入成功時，隱藏全畫面容器，遊戲畫面（#game-screen）會顯示
                document.getElementById("login-wrapper").style.display = "none";
                document.getElementById("game-screen").style.display = "flex";
                initGame();
            }
        }

        // 👇 加返 tryAgain 喺度，負責重新開始遊戲 👇
        function tryAgain() {
            let records = JSON.parse(localStorage.getItem("dailyRecords") || "{}");
            let playCount = records[today][empId] || 0;

            if (playCount >= MAX_PLAYS) {
                alert("你今日嘅 3 次機會已經用完啦，辛苦晒！");
                location.reload(); 
            } else {
                records[today][empId] = playCount + 1;
                localStorage.setItem("dailyRecords", JSON.stringify(records));
                
                document.getElementById("end-screen").style.display = "none";
                initGame();
            }
        }

        const canvas = document.getElementById("gameCanvas");
        const ctx = canvas.getContext("2d");
        
        let gameTimer = 40;
        let score = 0;
        let gameInterval;
        let animationId;
        let isPlaying = false;

        const goal = { x: 50, y: 30, width: 300, height: 40 }; 
        let baseGoalieSpeed = 1.5;
// 圖像大小設定：直接用 game 內部 canvas 座標控制，避免新 PNG 被自動比例壓細
const GOALIE_W = 95;
const GOALIE_H = 70;
const BLOCK_W = 58;
const BLOCK_H = 42;

let goalie = {
    x: 200 - GOALIE_W / 2,
    y: 22,
    width: GOALIE_W,
    height: GOALIE_H,
    speed: baseGoalieSpeed,
    targetX: 200 - GOALIE_W / 2
};
        
        // 人牆尺寸縮細；不用 imgBlock.onload 自動改高度，避免因圖片外圍留白而顯示比例失控
        let defenders = [
    { x: 95, y: 205, width: 58, height: 58, speed: 2, dir: 1 },
    { x: 235, y: 285, width: 58, height: 58, speed: 3, dir: -1 }
];

        let ball = { x: 200, y: 460, radius: 15, vx: 0, vy: 0, isShot: false };
        let target = { x: 200, y: 40 };
        let shooterX = 200; 

        document.getElementById("pos-slider").addEventListener("input", function(e) {
            if (!ball.isShot) {
                ball.x = parseInt(e.target.value);
                shooterX = ball.x; 
            }
        });

        canvas.addEventListener("mousedown", handleAimAndShoot);
        canvas.addEventListener("touchstart", function(e) {
            e.preventDefault();
            handleAimAndShoot(e.changedTouches[0]);
        }, {passive: false});

        function handleAimAndShoot(event) {
            if (ball.isShot || !isPlaying) return;
            const rect = canvas.getBoundingClientRect();
            
            // 支援手機 responsive canvas：將畫面座標轉回 canvas 內部 400x550 座標
            let tapX = (event.clientX - rect.left) * (canvas.width / rect.width);
            let tapY = (event.clientY - rect.top) * (canvas.height / rect.height);

            let dx = tapX - ball.x;
            let dy = tapY - ball.y;

            if (dy >= 0) { dy = -1; }

            let ratio = (0 - ball.y) / dy;
            let targetTopX = ball.x + (dx * ratio);

            if (targetTopX < 0) targetTopX = 0;
            if (targetTopX > canvas.width) targetTopX = canvas.width;

            let finalDx = targetTopX - ball.x;
            let finalDy = 0 - ball.y;
            let distance = Math.sqrt(finalDx * finalDx + finalDy * finalDy);
            
            let speed = 14; 
            ball.vx = (finalDx / distance) * speed;
            ball.vy = (finalDy / distance) * speed;
            ball.isShot = true;
        }

        function initGame() {
            clearInterval(gameInterval);
            cancelAnimationFrame(animationId);

            score = 0;
            baseGoalieSpeed = 1.5;
            gameTimer = 40; 
            isPlaying = true;
            document.getElementById("score-display").innerText = "🏆 分數: " + score;
            document.getElementById("level-display").innerText = "守門員速度: Lv 1";
            document.getElementById("time-display").innerText = "⏱️: " + gameTimer + "s";
            
            document.getElementById("pos-slider").disabled = false;

            gameInterval = setInterval(() => {
                gameTimer--;
                document.getElementById("time-display").innerText = "⏱️: " + gameTimer + "s";
                if (gameTimer <= 0) endGame();
            }, 1000);

            resetBall();
            gameLoop();
        }

        function showMessage(text, color) {
            let msg = document.getElementById("message-overlay");
            msg.innerText = text;
            msg.style.color = color;
            setTimeout(() => { msg.innerText = ""; }, 1000);
        }

        function resetBall() {
            ball.isShot = false;
            ball.vx = 0;
            ball.vy = 0;
            ball.y = 460;
            ball.x = parseInt(document.getElementById("pos-slider").value);
            shooterX = ball.x; 
        }

        function update() {
            if (!isPlaying) return;

            for (let d of defenders) {
                d.x += d.speed * d.dir;
                if (d.x <= 0 || d.x + d.width >= canvas.width) { d.dir *= -1; }
            }

            let distToTarget = goalie.targetX - goalie.x;
            if (Math.abs(distToTarget) < goalie.speed) {
                goalie.targetX = goal.x + Math.random() * (goal.width - goalie.width);
                goalie.speed = baseGoalieSpeed + Math.random() * 2; 
            } else {
                goalie.x += Math.sign(distToTarget) * goalie.speed;
            }

            if (ball.isShot) {
                ball.x += ball.vx;
                ball.y += ball.vy;

                for (let d of defenders) {
                    if (ball.y - ball.radius <= d.y + d.height && ball.y + ball.radius >= d.y &&
                        ball.x + ball.radius >= d.x && ball.x - ball.radius <= d.x + d.width) {
                        showMessage("Blocked!", "#FF9800"); 
                        resetBall();
                        return;
                    }
                }

                if (ball.y - ball.radius <= goalie.y + goalie.height && ball.y + ball.radius >= goalie.y &&
                    ball.x + ball.radius >= goalie.x && ball.x - ball.radius <= goalie.x + goalie.width) {
                    showMessage("Saved!", "#ff5252");
                    resetBall();
                    return;
                }

                if (ball.y <= goal.y + goal.height) {
                    if (ball.x >= goal.x && ball.x <= goal.x + goal.width) {
                        showMessage("Goal!!!", "#4CAF50");
                        score++;
                        baseGoalieSpeed += 0.5;
                        let currentLevel = Math.floor((baseGoalieSpeed - 1.5) / 0.5) + 1;
                        document.getElementById("score-display").innerText = "🏆 分數: " + score;
                        document.getElementById("level-display").innerText = "守門員速度: Lv " + currentLevel;
                    } else {
                        showMessage("You missed!", "#ff5252");
                    }
                    resetBall();
                }

                if (ball.x < 0 || ball.x > canvas.width || ball.y < 0) {
                    showMessage("You missed!", "#ff5252");
                    resetBall();
                }
            }
        }

        function drawAsset(img, x, y, w, h) {
            // 將圖片以 cover 方式畫入指定框，保持比例但自動裁走多餘邊位，
            // 適合用於有透明/留白邊位嘅角色 PNG。
            const imgRatio = img.naturalWidth / img.naturalHeight;
            const boxRatio = w / h;
            let sx = 0, sy = 0, sw = img.naturalWidth, sh = img.naturalHeight;

            if (imgRatio > boxRatio) {
                sw = img.naturalHeight * boxRatio;
                sx = (img.naturalWidth - sw) / 2;
            } else {
                sh = img.naturalWidth / boxRatio;
                sy = (img.naturalHeight - sh) / 2;
            }

            ctx.drawImage(img, sx, sy, sw, sh, x, y, w, h);
        }

        function draw() {
            // Canvas 內嘅球場背景：使用 field.jpg
            // 注意：bg.jpg 係 body / gameplay 外圍背景；field.jpg 先係球場畫布背景
            if (imgField.complete && imgField.naturalWidth !== 0) {
                ctx.drawImage(imgField, 0, 0, canvas.width, canvas.height);
            } else {
                const gradient = ctx.createLinearGradient(0, 0, 0, canvas.height);
                gradient.addColorStop(0, "#2E7D32");
                gradient.addColorStop(1, "#1B5E20");
                ctx.fillStyle = gradient;
                ctx.fillRect(0, 0, canvas.width, canvas.height);
            }

            // 保留一個簡潔嘅射門目標區，避免 field.jpg 本身太乾淨而玩家唔知射邊度
            ctx.strokeStyle = "rgba(255, 255, 255, 0.9)";
            ctx.lineWidth = 5;
            ctx.strokeRect(goal.x, goal.y, goal.width, goal.height);

            ctx.textAlign = "center";
            ctx.textBaseline = "middle";

            for (let d of defenders) {
    ctx.drawImage(
        imgBlock,
        d.x,
        d.y,
        d.width,
        d.height
    );
                } else {
                    ctx.fillStyle = "#FF9800"; 
                    ctx.fillRect(d.x, d.y, d.width, d.height);
                    
                    ctx.fillStyle = "black";
                    ctx.font = "25px sans-serif";
                    ctx.fillText("🧍‍♂️🧍‍♂️", d.x + d.width/2, d.y + d.height/2); 
                }
            }

            if (imgGoalie.complete && imgGoalie.naturalWidth !== 0) {
                drawAsset(imgGoalie, goalie.x, goalie.y, goalie.width, goalie.height);
            } else {
                ctx.fillStyle = "white"; 
                ctx.font = "40px sans-serif";
                ctx.fillText("🦸‍♂️", goalie.x + goalie.width/2, goalie.y + goalie.height/2);
            }

            if (imgShooter.complete && imgShooter.naturalWidth !== 0) {
                ctx.drawImage(imgShooter, shooterX - 25, 460 + 20, 50, 60);
            } else {
                ctx.fillStyle = "white"; 
                ctx.font = "45px sans-serif";
                ctx.fillText("🏃‍♂️", shooterX, 460 + 45); 
            }

            if (imgBall.complete && imgBall.naturalWidth !== 0) {
                ctx.drawImage(imgBall, ball.x - ball.radius, ball.y - ball.radius, ball.radius * 2, ball.radius * 2);
            } else {
                ctx.fillStyle = "white";
                ctx.font = "24px sans-serif";
                ctx.fillText("⚽", ball.x, ball.y);
            }
        }

        function gameLoop() {
            update();
            draw();
            if (isPlaying) { animationId = requestAnimationFrame(gameLoop); }
        }

        function endGame() {
            isPlaying = false;
            clearInterval(gameInterval);
            cancelAnimationFrame(animationId);
            document.getElementById("pos-slider").disabled = true;

            let records = JSON.parse(localStorage.getItem("dailyRecords") || "{}");
            let playCount = records[today][empId] || 0;
            let chancesLeft = MAX_PLAYS - playCount;

            document.getElementById("final-score").innerText = "你嘅分數係: " + score;
            let chancesText = document.getElementById("chances-left");
            let retryBtn = document.getElementById("retry-btn");
            let exitMsg = document.getElementById("exit-msg");
            
            if (chancesLeft > 0) {
                chancesText.innerText = `今日剩餘機會: ${chancesLeft} 次`;
                chancesText.style.color = "#4CAF50"; 
                retryBtn.style.display = "inline-block";
                if(exitMsg) exitMsg.style.display = "none";
            } else {
                chancesText.innerText = "今日 3 次機會已用完！";
                chancesText.style.color = "#ff5252"; 
                retryBtn.style.display = "none"; 
                if(exitMsg) exitMsg.style.display = "block";
            }

            document.getElementById("end-screen").style.display = "flex";

            let gameData = { staffId: empId, score: score };
            const GOOGLE_SHEET_URL = "https://script.google.com/macros/s/AKfycbzk0uSM-fh9nWZFnvJNwz8PmyupcIzZQPLc4Z3nJCQxoXM2dnnZg_tbzY9MeYxg_BOt/exec";

            fetch(GOOGLE_SHEET_URL, {
                method: "POST", mode: "no-cors", 
                headers: { "Content-Type": "text/plain;charset=utf-8" },
                body: JSON.stringify(gameData)
            }).then(() => { console.log("資料已發送！"); }).catch(error => { console.error("發送失敗：", error); });
        }
    </script>
